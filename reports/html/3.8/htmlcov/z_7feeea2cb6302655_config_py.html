1""" 

2Configuration Management Module for prismalog 

3 

4This module provides a centralized configuration system for the prismalog package, 

5enabling flexible and hierarchical configuration from multiple sources. It handles 

6loading, prioritizing, and accessing configuration settings throughout the application. 

7 

8Key features: 

9- Hierarchical configuration with clear priority order 

10- Multi-source configuration (default values, files, environment variables, CLI) 

11- Support for YAML configuration files 

12- Automatic type conversion for numeric and boolean values 

13- Command-line argument integration with argparse 

14- Singleton pattern to ensure configuration consistency 

15 

16The configuration system follows this priority order (highest to lowest): 

171. Programmatic settings via direct API calls or kwargs to initialize() 

182. Command-line arguments 

193. Configuration files (YAML) 

204. Environment variables (with support for CI/CD environments) 

215. Default values 

22 

23Environment Variables: 

24 Standard environment variables use the ``LOG_`` prefix: 

25 

26 - ``LOG_DIR``: Directory for log files 

27 - ``LOG_LEVEL``: Default logging level (e.g., INFO, DEBUG) 

28 - ``LOG_ROTATION_SIZE``: Size in MB for log rotation 

29 - ``LOG_BACKUP_COUNT``: Number of backup log files 

30 - ``LOG_FORMAT``: Log message format string 

31 - ``LOG_DATEFMT``: Log message date format string 

32 - ``LOG_FILENAME``: Base filename prefix for log files (default: 'app') 

33 - ``LOG_COLORED_CONSOLE``: Whether to use colored console output (true/false) 

34 - ``LOG_DISABLE_ROTATION``: Whether to disable log rotation (true/false) 

35 - ``LOG_EXIT_ON_CRITICAL``: Whether to exit on critical logs (true/false) 

36 - ``LOG_TEST_MODE``: Whether logger is in test mode (true/false) 

37 

38 For GitHub Actions, the same variables are supported with ``GITHUB_`` prefix: 

39 

40 - ``GITHUB_LOG_DIR``, ``GITHUB_LOG_LEVEL``, ``GITHUB_LOG_FILENAME``, ``GITHUB_LOG_DATEFMT``, etc. 

41 

42Command-Line Arguments: 

43 The following arguments can be parsed if `use_cli_args=True` during initialization: 

44 

45 - ``--log-config`` / ``--log-conf``: Path to logging configuration file (YAML) 

46 - ``--log-level`` / ``--logging-level``: Default logging level (DEBUG, INFO, etc.) 

47 - ``--log-dir`` / ``--logging-dir``: Directory for log files 

48 - ``--log-format`` / ``--logging-format``: Format string for log messages 

49 - ``--log-datefmt`` / ``--logging-datefmt``: Format string for log timestamps 

50 - ``--log-filename`` / ``--logging-filename``: Prefix for log filenames 

51 - ``--no-color`` / ``--no-colors``: Disable colored console output 

52 - ``--disable-rotation``: Disable log file rotation 

53 - ``--exit-on-critical``: Exit the program on critical errors 

54 - ``--rotation-size``: Log file rotation size in MB 

55 - ``--backup-count``: Number of backup log files to keep 

56 

57Type Conversion: 

58 String values from configuration files and environment variables are automatically 

59 converted to the appropriate types (boolean, integer) based on the default config's 

60 type definition. Boolean values support multiple string representations: 

61 - True: "true", "1", "yes", "y", "t", "on" 

62 - False: "false", "0", "no", "n", "f", "off", "none" 

63 

64Usage examples: 

65 # Basic initialization with defaults 

66 LoggingConfig.initialize() 

67 

68 # Initialization with configuration file and CLI args enabled 

69 LoggingConfig.initialize(config_file="logging_config.yaml", use_cli_args=True) 

70

72 log_dir = LoggingConfig.get("log_dir") 

73 filename_prefix = LoggingConfig.get_filename_prefix() 

74 

75 # Setting configuration values programmatically 

76 LoggingConfig.set("colored_console", False) 

77 

78 # Getting appropriate log level for a specific logger 

79 level = LoggingConfig.get_level("requests.packages.urllib3")

185 config_sources = cls._collect_configurations(config_file, use_cli_args, kwargs)

188 cls._apply_configurations(config_sources)

192 @classmethod 

193 def _collect_configurations( 

194 cls, config_file: Optional[str], use_cli_args: bool, kwargs: Dict[str, Any] 

195 ) -> Dict[str, Dict[str, Any]]: 

196 """ 

197 Collect configurations from all possible sources and convert types immediately. 

198 

199 This method gathers configuration from multiple sources: 

200 - Default values 

201 - Environment variables 

202 - Configuration files (if specified) 

203 - Command-line arguments (if use_cli_args is True) 

204 - Direct keyword arguments 

205 

206 Each source's values are converted to appropriate types during collection. 

207 

208 Args: 

209 config_file: Optional path to a configuration file 

210 use_cli_args: Whether to parse command-line arguments 

211 kwargs: Direct configuration values passed to initialize() 

212 

213 Returns: 

214 Dictionary containing configuration values from each source 

215 """ 

216 # Start with empty collections for each source 

217 sources: Dict[str, Dict[str, Any]] = { 

218 "defaults": cls.DEFAULT_CONFIG.copy(), 

219 "file": {}, 

220 "env": {}, 

221 "cli": {}, 

222 "kwargs": kwargs.copy() if kwargs else {}, 

223 } 

224 

225 # Collect and convert file configuration 

226 if config_file and os.path.exists(config_file): 

227 raw_file_config = cls._load_raw_file_config(config_file) 

228 if raw_file_config: 

229 # Convert types immediately 

230 file_config = cls._convert_config_values(raw_file_config) 

231 sources["file"] = file_config 

232 cls.debug_print(f"Collected from file: {file_config}") 

233 

234 # Collect and convert environment variables 

235 raw_env_config = cls._load_raw_env_config() 

236 if raw_env_config: 

237 # Convert types immediately 

238 env_config = cls._convert_config_values(raw_env_config) 

239 sources["env"] = env_config 

240 cls.debug_print(f"Collected from environment: {env_config}") 

241 

242 # Collect and convert command line arguments 

243 if use_cli_args: 

244 raw_arg_config = cls._load_raw_cli_args() 

245 if raw_arg_config: 

246 # Convert types immediately 

247 arg_config = cls._convert_config_values(raw_arg_config) 

248 sources["cli"] = arg_config 

249 cls.debug_print(f"Collected from CLI: {arg_config}") 

250 

251 return sources

253 @classmethod 

254 def _apply_configurations(cls, sources: Dict[str, Dict[str, Any]]) -> None: 

255 """ 

256 Apply configurations in the correct priority order. 

257 

258 Priority order (lowest to highest): 

259 1. Default values (starting point) 

260 2. Environment variables (override defaults) 

261 3. Configuration file (overrides environment) 

262 4. Command-line arguments (overrides file) 

263 5. Direct keyword arguments (highest priority) 

264 

265 Args: 

266 sources: Dictionary containing configurations from different sources 

267 """ 

268 # 1. Start with defaults 

269 cls._config = sources["defaults"].copy() 

270 cls.debug_print(f"1. Starting with defaults: {cls._config}") 

271 

272 # 2. Apply environment variables (overrides defaults) 

273 if sources["env"]: 

274 cls.debug_print("2. Applying environment variables") 

275 cls._config.update(sources["env"]) 

276 cls.debug_print(f" Config after env vars: {cls._config}") 

277 

278 # 3. Apply configuration file (overrides env) 

279 if sources["file"]: 

280 cls.debug_print("3. Applying file configuration") 

281 cls._config.update(sources["file"]) 

282 cls.debug_print(f" Config after file: {cls._config}") 

283 

284 # 4. Apply command-line arguments (overrides file) 

285 if sources["cli"]: 

286 cls.debug_print("4. Applying command line arguments") 

287 cls._config.update(sources["cli"]) 

288 cls.debug_print(f" Config after CLI args: {cls._config}") 

289 

290 # 5. Apply direct kwargs (highest priority) 

291 if sources["kwargs"]: 

292 cls.debug_print("5. Applying kwargs") 

293 cls._config.update(sources["kwargs"]) 

294 cls.debug_print(f" Config after kwargs: {cls._config}") 

295 

296 cls._initialized = True 

297 cls.debug_print(f"Final configuration: {cls._config}")

299 @classmethod 

300 def _convert_config_values(cls, config_dict: Dict[str, Any]) -> Dict[str, Any]: 

301 """ 

302 Convert string configuration values to their appropriate types. 

303 

304 This is the central type conversion method used by all config sources. 

305 It handles two types of conversions: 

306 1. Boolean values: Converts strings like "true", "yes", "1" to True and their opposites to False 

307 2. Numeric values: Converts digit strings to integers 

308 

309 The method processes boolean conversions first, then numeric conversions. 

310 Invalid conversion attempts result in the key being removed from the result. 

311 

312 Args: 

313 config_dict: Dictionary containing configuration values to convert 

314 

315 Returns: 

316 Dictionary with values converted to appropriate types 

317 

318 Examples: 

319 >>> LoggingConfig._convert_config_values({"colored_console": "true", "rotation_size_mb": "50"}) 

320 {'colored_console': True, 'rotation_size_mb': 50} 

321 """ 

322 if not config_dict: 

323 return {} 

324 

325 result = config_dict.copy() 

326 

327 # Get the keys and their types from DEFAULT_CONFIG 

328 numeric_keys = [k for k, v in cls.DEFAULT_CONFIG.items() if isinstance(v, int)] 

329 boolean_keys = [k for k, v in cls.DEFAULT_CONFIG.items() if isinstance(v, bool)] 

330 

331 # First, convert boolean values 

332 for key in boolean_keys: 

333 if key in result and isinstance(result[key], str): 

334 val = result[key].strip().lower() 

335 if val in ["true", "1", "yes", "y", "t", "on"]: 

336 result[key] = True 

337 cls.debug_print(f"Converted '{key}={val}' to boolean True") 

338 elif val in ["false", "0", "no", "n", "f", "off", "none"]: 

339 result[key] = False 

340 cls.debug_print(f"Converted '{key}={val}' to boolean False") 

341 else: 

342 cls.debug_print(f"Warning: Cannot convert '{key}={val}' to boolean") 

343 result.pop(key, None) # Remove invalid values 

344 

345 # Then, convert numeric values 

346 for key in numeric_keys: 

347 if key in result and isinstance(result[key], str): 

348 val = result[key].strip() 

349 if val.isdigit() or (val.startswith("-") and val[1:].isdigit()): 

350 result[key] = int(val) 

351 cls.debug_print(f"Converted '{key}={val}' to integer {int(val)}") 

352 else: 

353 cls.debug_print(f"Warning: Cannot convert '{key}={val}' to integer") 

354 result.pop(key, None) # Remove invalid values 

355 

356 return result

358 @classmethod 

359 def _load_raw_file_config(cls, config_path: str) -> Dict[str, Any]: 

360 """ 

361 Load raw configuration from file without type conversion. 

362 

363 Supports YAML file format. If PyYAML is not installed, 

364 YAML files cannot be loaded and an empty dictionary is returned. 

365 

366 Args: 

367 config_path: Path to the configuration file 

368 

369 Returns: 

370 Dictionary with raw configuration values from the file, 

371 or empty dictionary if file doesn't exist or has invalid format 

372 """ 

373 file_config: Dict[str, Any] = {} # Add type annotation here 

374 if not os.path.exists(config_path): 

375 return file_config 

376 

377 try: 

378 with open(config_path, mode="r", encoding="utf-8") as f: 

379 if config_path.endswith((".yaml", ".yml")): 

380 try: 

381 import yaml 

382 

383 file_config = yaml.safe_load(f) 

384 except ImportError: 

385 print("YAML configuration requires PyYAML. Install with: pip install PyYAML") 

386 print("Continuing with default configuration.") 

387 return file_config 

388 else: 

389 cls.debug_print(f"Unsupported config file format: {config_path}") 

390 return file_config 

391 

392 except Exception as e: 

393 cls.debug_print(f"Error loading config file: {e}") 

394 return file_config 

395 

396 # Process level values to ensure they are uppercase 

397 if "default_level" in file_config and isinstance(file_config["default_level"], str): 

398 file_config["default_level"] = file_config["default_level"].upper() 

399 

400 # Also handle external_loggers levels 

401 if "external_loggers" in file_config and isinstance(file_config["external_loggers"], dict): 

402 for logger, level in file_config["external_loggers"].items(): 

403 if isinstance(level, str): 

404 file_config["external_loggers"][logger] = level.upper() 

405 

406 return file_config

408 @classmethod 

409 def _load_raw_env_config(cls) -> Dict[str, Any]: 

410 """ 

411 Load raw environment variables without type conversion. 

412 

413 Looks for environment variables with both LOG_ and GITHUB_ prefixes. 

414 For each configuration key, it checks the variables in order and uses 

415 the first one found. 

416 

417 Returns: 

418 Dictionary mapping configuration keys to environment variable values 

419 """ 

420 env_config = {} 

421 

422 # Define lookup tables with ordered priorities 

423 env_vars = { 

424 "log_dir": ["LOG_DIR", "GITHUB_LOG_DIR"], 

425 "default_level": ["LOG_LEVEL", "GITHUB_LOG_LEVEL"], 

426 "rotation_size_mb": ["LOG_ROTATION_SIZE", "GITHUB_LOG_ROTATION_SIZE"], 

427 "backup_count": ["LOG_BACKUP_COUNT", "GITHUB_LOG_BACKUP_COUNT"], 

428 "log_format": ["LOG_FORMAT", "GITHUB_LOG_FORMAT"], 

429 "datefmt": ["LOG_DATEFMT", "GITHUB_LOG_DATEFMT"], 

430 "log_filename": ["LOG_FILENAME", "GITHUB_LOG_FILENAME"], 

431 "colored_console": ["LOG_COLORED_CONSOLE", "GITHUB_LOG_COLORED_CONSOLE"], 

432 "disable_rotation": ["LOG_DISABLE_ROTATION", "GITHUB_LOG_DISABLE_ROTATION"], 

433 "exit_on_critical": ["LOG_EXIT_ON_CRITICAL", "GITHUB_LOG_EXIT_ON_CRITICAL"], 

434 "test_mode": ["LOG_TEST_MODE", "GITHUB_LOG_TEST_MODE"], 

435 } 

436 

437 # Efficiently check each config key using direct lookup 

438 for config_key, env_vars_list in env_vars.items(): 

439 for env_var in env_vars_list: 

440 if env_var in os.environ: 

441 env_config[config_key] = os.environ[env_var] 

442 break # Stop after finding the first matching env var 

443 

444 return env_config

643 Args: 

644 key: The configuration key to retrieve 

645 default: Value to return if the key is not found 

646 

647 Returns: 

648 The configuration value or the default if not found 

649 

650 Example: 

651 ```python 

652 log_dir = LoggingConfig.get("log_dir") 

653 level = LoggingConfig.get("default_level", "INFO") 

654 ``` 

655 """ 

656 if not cls._initialized: 

657 cls.initialize() 

658 

659 # Special handling for nested keys 

660 if key == "module_levels": 

661 # Return the module_levels dictionary if it exists 

662 if "module_levels" in cls._config and isinstance(cls._config["module_levels"], dict): 

663 return cls._config["module_levels"] 

664 return {} 

665 

666 # Normal key 

667 # First check if it's in current config 

668 if key in cls._config: 

669 return cls._config[key] 

670 

671 # If not, check if it's in DEFAULT_CONFIG 

672 if key in cls.DEFAULT_CONFIG: 

673 return cls.DEFAULT_CONFIG[key] 

674 

675 # If not found in either place, return the provided default 

676 return default

678 @classmethod 

679 def set(cls, key: str, value: Any) -> None: 

680 """ 

681 Set or update a configuration value. 

682 

683 This allows runtime modification of configuration values. Changes 

684 will be reflected in any new loggers created after the change. 

685 

686 Supports nested keys with dot notation (e.g., "module_levels.my_module"). 

687 

688 Args: 

689 key: The configuration key to set 

690 value: The value to assign to the key 

691 

692 Example: 

693 >>> # Disable exiting on critical logs 

694 >>> LoggingConfig.set("exit_on_critical", False) 

695 >>> 

696 >>> # Set module-specific log level 

697 >>> LoggingConfig.set("module_levels.my_module", "DEBUG") 

698 """ 

699 # Handle nested keys (with dot notation) 

700 if "." in key: 

701 # Split the key path 

702 parts = key.split(".") 

703 main_key = parts[0] 

704 sub_key = parts[1] 

705 

706 # Ensure parent dictionary exists 

707 if main_key not in cls._config: 

708 cls._config[main_key] = {} 

709 elif not isinstance(cls._config[main_key], dict): 

710 # If it exists but is not a dict, convert it to a dict 

711 cls._config[main_key] = {} 

712 

713 # Set the value in the nested dict 

714 cls._config[main_key][sub_key] = value 

715 else: 

716 # Simple case: direct key 

717 cls._config[key] = value

719 @classmethod 

720 def get_level(cls, name: Optional[str] = None, default_level: Optional[str] = None) -> int: 

721 """ 

722 Get the numeric log level based on configuration priority. 

723 

724 This method determines the appropriate log level based on priority order: 

725 1. Explicit level parameter (highest priority) 

726 2. Logger-specific configuration from external_loggers 

727 3. Default level from configuration 

728 

729 Args: 

730 name: Optional logger name to check for specific configuration 

731 default_level: Optional override for the default level 

732 

733 Returns: 

734 The numeric logging level (from logging module constants) 

735 

736 Examples: 

737 Get default level for application: 

738 

739 >>> default_level = LoggingConfig.get_level() 

740 

741 Get level for a specific module: 

742 

743 >>> requests_level = LoggingConfig.get_level("requests.packages.urllib3") 

744 

745 Override with explicit level: 

746 

747 >>> debug_level = LoggingConfig.get_level(default_level="DEBUG") 

748 """ 

749 # 1. First priority: Use explicit level if provided 

750 if default_level: 

751 return cls.map_level(default_level) 

752 

753 # 2. Second priority: Check logger-specific config 

754 if name: 

755 external_loggers = cls.get("external_loggers", {}) 

756 if name in external_loggers: 

757 return cls.map_level(external_loggers[name]) 

758 

759 # 3. Third priority: Use configured default level 

760 config_level = cls.get("default_level", "INFO") 

761 return cls.map_level(config_level)
